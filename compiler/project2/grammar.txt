MATH -> A
A -> A + B @add
A -> A - B @minus
A -> B
B -> B * C @multiply
B -> B / C @divide
B -> C
C -> - C @invert
C -> D
D -> E ^ D @pow
D -> E
E -> TERM
E -> ( MATH )
